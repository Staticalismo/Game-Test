    var svg = d3.select("svg"),
        width = +svg.attr("width"),
        height = +svg.attr("height");

    var radius = 5;
    // generate first 3 point
    var circle_data = d3.range(3).map(function () {
        return {
            x: Math.round(Math.random() * (width - radius * 2 ) + radius),
            y: Math.round(Math.random() * (height - radius * 2 ) + radius)
        };
    });

    //define my line generator
    var bezierLine = d3.line()
        .x(function (d) {
            return d.x;
        })
        .y(function (d) {
            return d.y;
        }).curve(d3.curveCardinal);


    d3.select("svg").append("path")
        .attr("stroke", "blue")
        .attr("stroke-linecap", "round")
        .attr("stroke-width", 5)
        .attr("fill", "none")
        .attr("d", bezierLine(circle_data)).transition()
        .duration(2000)
        .attrTween("stroke-dasharray", function () {
            var len = this.getTotalLength();
            return function (t) {
                return (d3.interpolateString("0," + len, len + ",0"))(t)
            };
        });
		// add circle
    var circles = d3.select("svg")
        .append("g")
        .attr("class", "circles")
        .selectAll("circle")
        .data(circle_data)
        .enter()
        .append("circle")
        .attr("cx", function (d) {
            return (d.x)
        })
        .attr("cy", function (d) {
            return (d.y)
        })
        .attr("r", radius)
        .attr("fill",d3.interpolateRainbow(Math.random()))

    var prevlen = 0;


    svg.on("click", function (d) {
        // add a nex anchor point
        circle_data.push({
            x: d3.event.x,
            y: d3.event.y
        });
        
        // transition the old curve
        d3.select("path")
            .each(function (d) {
                // get previous length of the path
                prevlen = this.getTotalLength();
            })
            .transition()
            .duration(2000)
            .attr("d", bezierLine(circle_data))

        // transition to new point usith strok dasharray trick
        d3.select("path")
            .transition()
            .duration(2000)
            .attr("d", bezierLine(circle_data))
            .attrTween("stroke-dasharray", function () {
                var len = this.getTotalLength();
                return function (t) {
                    return (d3.interpolateString(prevlen + " ," + (len - prevlen), (len+50 )+ ",50"))(t)
                };
                
            });
        // add new point as a circle
        svg.selectAll("circle")
            .data(circle_data)
            .enter()
            .append("circle")
            .attr("cx", function (d) {
                return (d.x)
            })
            .attr("cy", function (d) {
                return (d.y)
            })
            .attr("r", radius)
		        .attr("fill",d3.interpolateRainbow(Math.random()))
    })
